def solution(prices, algo, k):
    cur = sum([i*(j-0.5)*2 for i, j in zip(prices, algo)])
    
    start = cur
    for i in range(k):
        if algo[i]==0:
            start += prices[i]*2
    best = start
    for i in range(1,len(algo)-k+1):
        
        if algo[i+k-1]==0:
            start += prices[i+k-1]*2
        if algo[i-1]==0:
            start -= prices[i-1]*2
        if start > best:
            best = start
        
    return int(best)
