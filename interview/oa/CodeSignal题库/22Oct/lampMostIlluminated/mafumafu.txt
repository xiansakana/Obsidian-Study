import collections
def solution(lamps):
    dc = collections.defaultdict(int)
    for i in range(len(lamps)):
        
        dc[lamps[i][0] - lamps[i][1]] += 1
        dc [lamps[i][0] + lamps[i][1] + 1] -=1
        
    arr = sorted(dc.keys())
    best_pos = 0
    best_ilum = 0
    cur = 0
    
    for i in arr:
        cur = dc[i] + cur
        if cur>best_ilum:
            best_ilum = cur
            best_pos = i
    return best_pos
